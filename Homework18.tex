\begin{problem}{1}
You can use the distributive property to write \underline{equivalent} expressions.
\end{problem}

\begin{problem}{2}
\begin{align*}
    &\text{Distributive Property}       &\text{Check} \\
    &5(3 + 7)        &5(3 + 7) \\
    = &5(3) + 5(7)   &=5(10) \\
    = &15 + 35       &=\underline{50.} \\
    = &\boxed{50.}   &
\end{align*}
\end{problem}
